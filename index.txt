function findEntry(q) {
  const l = q.toLowerCase();
  for (const [,entry] of Object.entries(KB)) {
    if (entry.key.some(k => l.includes(k))) return entry;
  }
  return null;
}

function render(entry, q) {
  if (!entry) {
    return `Goede vraag over "<b>${q}</b>"! Ik heb hier nu geen direct antwoord op in mijn kennisbank.<br><br>Neem contact op met je <b>teamleider</b> of <b>buddy</b> voor het juiste antwoord. Die helpen je graag verder! 😊<br><br><i style="font-size:12px;color:var(--subtiel)">Tip: probeer je vraag anders te formuleren — misschien kan ik je dan toch helpen.</i>`;
  }
  let h = `<b>${entry.title}</b><br><br>${entry.intro}<br>`;
  if (entry.items) {
    h += `<ul style="margin:8px 0 4px 16px">`;
    entry.items.forEach(i => { h += `<li style="margin-bottom:5px">${i}</li>`; });
    h += `</ul>`;
  }
  if (entry.sections) {
    entry.sections.forEach(s => { h += `<span class="sec-title">${s.label}</span>${s.text}<br>`; });
  }
  if (entry.tags) {
    h += `<br>`;
    entry.tags.forEach(t => { h += `<span class="tag">${t}</span>`; });
    h += `<br>`;
  }
  if (entry.tip) {
    h += `<div class="tip-box">${entry.tip}</div>`;
  }
  return h;
}

let busy = false;
function getTime() { return new Date().toLocaleTimeString('nl-NL',{hour:'2-digit',minute:'2-digit'}); }
function resize(el) { el.style.height='auto'; el.style.height=Math.min(el.scrollHeight,90)+'px'; }
function onKey(e) { if(e.key==='Enter'&&!e.shiftKey){e.preventDefault();sendMsg();} }
function scrollBot() { const c=document.getElementById('chat'); setTimeout(()=>{c.scrollTop=c.scrollHeight;},50); }

function addMsg(role, html) {
  const c = document.getElementById('chat');
  const isUser = role==='user';
  const d = document.createElement('div');
  d.className = `msg ${isUser?'user':'bot'}`;
  d.innerHTML = `
    <div class="av ${isUser?'user':'bot'}">${isUser?'👤':'🗺️'}</div>
    <div class="bubble-wrap">
      <div class="bubble">${html}</div>
      ${!isUser?`<div class="feedback">
        <button class="fb" onclick="fb(this,'👍')">👍 Nuttig</button>
        <button class="fb" onclick="fb(this,'👎')">👎 Niet handig</button>
      </div>`:''}
      <div class="ts">${getTime()}</div>
    </div>`;
  c.appendChild(d);
  scrollBot();
}

function fb(btn, type) {
  btn.parentElement.querySelectorAll('.fb').forEach(b=>{b.classList.remove('on');});
  btn.classList.add('on');
  btn.textContent = type==='👍' ? '👍 Bedankt!' : '👎 Noteer ik!';
}

}

function sendMsg() {
  const inp=document.getElementById('inp');
  const txt=inp.value.trim();
  if(!txt||busy) return;
  busy=true;
  inp.value=''; inp.style.height='auto';
  document.getElementById('sendBtn').disabled=true;
  addMsg('user', txt);
  showTyping();
  setTimeout(()=>{
    hideTyping();
